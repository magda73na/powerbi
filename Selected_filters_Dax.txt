Selected Filters = 
VAR SelStatus =
    VAR IsFilteredStatus = ISFILTERED(D_Status[StatusName])
    VAR CountStatus = COUNTROWS(VALUES(D_Status[StatusName]))
    RETURN
        SWITCH(
            TRUE(),
            NOT IsFilteredStatus, "All selected",
            CountStatus > 3, "Above 3",
            CONCATENATEX(VALUES(D_Status[StatusName]), D_Status[StatusName], ", ")
        )
